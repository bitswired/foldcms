## See It In Action

<Tabs>
  <TabItem label="Define Schema">
    <Code code={`import { Schema } from "effect";

const PostSchema = Schema.Struct({
  id: Schema.String,
  title: Schema.String,
  authorId: Schema.String,
  tags: Schema.Array(Schema.String),
  publishedAt: Schema.DateFromString,
});`} lang="typescript" title="schemas.ts" theme="dracula" />
  </TabItem>
  
  <TabItem label="Create Collection">
    <Code code={`import { defineCollection } from "@foldcms/core";
import { jsonFilesLoader } from "@foldcms/core/loaders";

const posts = defineCollection({
  loadingSchema: PostSchema,
  loader: jsonFilesLoader(PostSchema, {
    folder: "content/posts"
  }),
  relations: {
    authorId: {
      type: "single",
      field: "authorId", 
      target: "authors"
    }
  }
});`} lang="typescript" title="collections.ts" theme="dracula" />
  </TabItem>
  
  <TabItem label="Query Content">
    <Code code={`const cms = yield* CmsTag;

// Get all posts - fully typed!
const posts = yield* cms.getAll("posts");
//    ^? readonly Post[]

// Get specific post
const post = yield* cms.getById("posts", "hello-world");
//    ^? Option<Post>

// Load relations with type safety
const author = yield* cms.loadRelation("posts", post, "authorId");
//    ^? Option<Author>`} lang="typescript" title="queries.ts" theme="dracula" />
  </TabItem>

## Three Types of Relations

FoldCMS supports powerful, type-safe relationships between your content:

<CardGrid>
  <Card title="Single Relations" icon="right-arrow">
    One-to-one relationships. Returns `Option<T>`.
    
    ```typescript
    relations: {
      authorId: {
        type: "single",
        field: "authorId",
        target: "authors"
      }
    }
    ```
  </Card>
  
  <Card title="Array Relations" icon="list-format">
    One-to-many relationships. Returns `readonly T[]`.
    
    ```typescript
    relations: {
      tagIds: {
        type: "array",
        field: "tagIds",
        target: "tags"
      }
    }
    ```
  </Card>
  
  <Card title="Map Relations" icon="puzzle">
    Key-value relationships. Returns `ReadonlyMap<string, T>`.
    
    ```typescript
    relations: {
      translations: {
        type: "map",
        field: "i18nMap",
        target: "content"
      }
    }
    ```
  </Card>
</CardGrid>

## Quick Start

<Code code={`# Install FoldCMS
bun add @foldcms/core effect @effect/platform @effect/sql-sqlite-bun

# Create your first collection
import { defineCollection, makeCms, build } from "@foldcms/core";
import { jsonFilesLoader } from "@foldcms/core/loaders";

const posts = defineCollection({
  loadingSchema: PostSchema,
  loader: jsonFilesLoader(PostSchema, { folder: "content/posts" })
});

const { CmsTag, CmsLayer } = makeCms({ collections: { posts } });

// Build and query
yield* build({ collections: { posts } });
const cms = yield* CmsTag;
const allPosts = yield* cms.getAll("posts");`} lang="typescript" title="Get started in minutes" theme="dracula" />
